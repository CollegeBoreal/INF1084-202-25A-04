# Script : services1.ps1
# Objectif : Lister tous les services Active Directory et vérifier leur état

# Lister les services liés à AD
Get-Service | Where-Object {
    $_.DisplayName -like "*Directory*" -or $_.Name -match "NTDS|ADWS|DFSR|kdc|Netlogon|IsmServ"
} | Sort-Object DisplayName

# Vérifier l’état des services principaux
Get-Service -Name NTDS, ADWS, DFSR, KDC, Netlogon, IsmServ
